ASG_military_update_stats = {
	every_country = {
		set_variable = {
			ASG_industrial_score = THIS.num_of_military_factories
		}
		set_variable = {
			ASG_industrial_score_modifier = modifier@industrial_capacity_factory
		}
		add_to_variable = {
			ASG_industrial_score_modifier = 1
		}
		clamp_variable = {
			var = ASG_industrial_score_modifier
			min = 0
			max = 10000
		}
		multiply_variable = {
			ASG_industrial_score = ASG_industrial_score_modifier
		}
		# log = "Score = [?ASG_industrial_score] THIS = [?THIS.GetName] ROOT = [?ROOT.GetName]"
	}
}

ASG_equipment_update_stats = {
	every_country = {
		### only count equipment that has a major impact in combat
		set_temp_variable = {
			fielded_eq_ratio = num_equipment_in_armies_k@infantry_equipment
		}
		divide_temp_variable = {
			fielded_eq_ratio = num_target_equipment_in_armies_k@infantry_equipment
		}
		set_temp_variable = {
			tot_fielded_eq_ratio = 0
		}
		add_to_temp_variable = {
			tot_fielded_eq_ratio = fielded_eq_ratio
		}
		set_temp_variable = {
			equipment_count = 0
		}
		add_to_temp_variable = {
			equipment_count = 1
		}
		if = {
			limit = {
				check_variable = {
					num_target_equipment_in_armies_k@artillery_equipment > 0.25
				}
			}
			set_temp_variable = {
				fielded_eq_ratio = num_equipment_in_armies_k@artillery_equipment
			}
			divide_temp_variable = {
				fielded_eq_ratio = num_target_equipment_in_armies_k@artillery_equipment
			}
			add_to_temp_variable = {
				tot_fielded_eq_ratio = fielded_eq_ratio
			}
			add_to_temp_variable = {
				equipment_count = 1
			}
		}
		if = {
			limit = {
				check_variable = {
					num_target_equipment_in_armies_k@support_equipment > 0.25
				}
			}
			set_temp_variable = {
				fielded_eq_ratio = num_equipment_in_armies_k@support_equipment
			}
			divide_temp_variable = {
				fielded_eq_ratio = num_target_equipment_in_armies_k@support_equipment
			}
			add_to_temp_variable = {
				tot_fielded_eq_ratio = fielded_eq_ratio
			}
			add_to_temp_variable = {
				equipment_count = 1
			}
		}
		###
		divide_temp_variable = {
			tot_fielded_eq_ratio = equipment_count
		}
		multiply_temp_variable = {
			tot_fielded_eq_ratio = 100
		}
		set_variable = {
			ASG_equipment_score = tot_fielded_eq_ratio
		}
	}
}

	multiply_variable = {
		GLOBAL.ASG_military_scale_factor = 2
	}
	for_each_loop = {
		array = GLOBAL.ASG_SAMPLE_military_graph_offsets_blue
		set_temp_variable = {
			value_slope = v
		}
		divide_temp_variable = {
			value_slope = 2
		}
		set_variable = {
			GLOBAL.ASG_SAMPLE_military_graph_offsets_blue^i = value_slope
		}
		if = {
			limit = {
				check_variable = {
					i > 0
				}
			}
			set_temp_variable = {
				index = i
			}
			subtract_from_temp_variable = {
				index = 1
