on_actions = {
	on_startup = {
		effect = {
			SER = {
				country_event = { id = ww1_franzferdinand.2 days = 1304 }
			}
			every_country={
				limit={
					is_playable_country=yes
				}
				add_ideas = peace_training
				set_variable = { super_events_enabled = 2 }
			}
		}
	}
	on_monthly = {
		effect = {
			every_country={
				limit = { 
					is_playable_country=yes
					NOT = {
						OR = {
							has_war = yes
							has_idea = peace_training 
						}
					} 
				}
				add_ideas = peace_training
			}
		}
	}
	on_capitulation = {
		effect = { 
			if = {
				limit = { 
					NOT = { has_global_flag = balkan_war_over }
					original_tag = SER 
					TUR = { has_war_with = SER }
				}
				set_global_flag = balkan_war_over
				TUR = { 
					country_event = { id = ww1_firstbalkan.200 } 
					TUR = { white_peace = SER }		
					SER = { white_peace = TUR }
					BUL = { white_peace = TUR }
					GRE = { white_peace = TUR }
					MNT = { white_peace = TUR }
				}
			}
		}
	}
}
